Using Code to Cause an Input Event to Fire an Output Event

If you create a function block and link one of its input events to an output event using C# code, and also use the EcoStruxure Automation Expert Local HMI tool to run multiple canvases, the output event can fire both when the input event is triggered and when the Local HMI tool switches focus to the canvas that is the target of the C# code.

For example, consider the following iThis function block.

Code that Fires an Output Event when the Local HMI Brings Focus to the Canvas

If the following code is implemented for the iTHIS function block, the TestEventHandler output event is fired both when the ReqEventHandler input is triggered and also when the EcoStruxure Automation Expert Local HMI brings focus to the canvas : 

Public partial class sDefault : NxtControl.GuiFramework.HMISymbol
{
    public sDefault()
    {
         //
         //
The IntializeComponent() call is required for Windws Forms designer support.
         //
         InitializeComponent();
         this.ReqEventHandler_Fired += ReqEventHandler_Fired_EventHandler;
    }
    void ReqEventHandler_Fired_EventHandler(object sender,ReqEventHandlerEventArgs e)
    {
        FireEvent_TestEventHandler(5);
    }

Code that Fires an Output Event only when an Input Event is Triggered

To avoid the unintended firing of the TestEventHandler output event, the following code fires the TestEventHandler output event only when the ReqEventHandler input is fired and not when the EcoStruxure Automation Expert Local HMI brings focus to the canvas:: 

using System;
using System.Reflection;
using NxtControl.Services;
using NxtControl.GuiFramework;
using NxtControl.Communication;

namespace HMI.Main.Symbols.CAT1
{
	/// <summary>
///	Description of sDefault.
///	</summary>
public partial class sDefault : NxtControl.GuiFramework.HMISymbol
	{
		public sDefault()
		{
			//
			// The InitializeComponent() call is required for Windows Forms designer support.
			//
			InitializeComponent();
			InitializeReflection();
			this.DEM_SAISIE_LOT_Fired += DEM_SAISIE_LOT_Fired_EventHandler;
		}

		private static FieldInfo channelIdInfo;
		private static FieldInfo cookieInfo;
		private static FieldInfo eventIndexInfo;
		private static IHMIAccessorService accessorService;
		private static void InitializeReflection()
		{
			if (channelIdInfo == null)
			{
				channelIdInfo = typeof(DEM_SAISIE_LOTEventArgs).GetField("channelId", BindingFlags.Instance | BindingFlags.GetField | BindingFlags.NonPublic);
				cookieInfo = typeof(DEM_SAISIE_LOTEventArgs).GetField("cookie", BindingFlags.Instance | BindingFlags.GetField | BindingFlags.NonPublic);
				eventIndexInfo = typeof(DEM_SAISIE_LOTEventArgs).GetField("eventIndex", BindingFlags.Instance | BindingFlags.GetField | BindingFlags.NonPublic);
                accessorService = (IHMIAccessorService)ServiceProvider.GetService(typeof(IHMIAccessorService));
                LoggingService.Info("setup:" + (channelIdInfo == null ? "failed" : "success"));
			}

		}
		
		void DEM_SAISIE_LOT_Fired_EventHandler(object sender, DEM_SAISIE_LOTEventArgs e)
		{
			LoggingService.Info(channelIdInfo.GetValue(e));
			LoggingService.Info(cookieInfo.GetValue(e));
			LoggingService.Info(eventIndexInfo.GetValue(e));
		
            EventStatusInformation eventStatusInformation = null;
            accessorService.GetEventStatusValue((int)channelIdInfo.GetValue(e), (int)cookieInfo.GetValue(e), (int) eventIndexInfo.GetValue(e), true, 0, ref eventStatusInformation);
            LoggingService.Info(eventStatusInformation.Valid);
            LoggingService.Info(eventStatusInformation.Active);
            LoggingService.Info(eventStatusInformation.Initial);
            if(!eventStatusInformation.Initial)
            {
                // fire output event here
                FireEvent_TestEventHandler(0, true);
             }
			
		}
	}
}
